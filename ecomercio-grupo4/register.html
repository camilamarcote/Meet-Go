const API_URL = "https://api.meetandgouy.com"; document.addEventListener("DOMContentLoaded", async () => { // 🔐 EXTRAER SESIÓN CORRECTAMENTE DESDE TU OBJETO BASE const authUser = JSON.parse(localStorage.getItem("currentUser")); const token = authUser?.token; if (!token) { console.warn("⚠️ No se detectó token de sesión activo. Redirigiendo..."); window.location.href = "login.html"; return; } const form = document.getElementById("profileForm"); if (!form) { console.error("❌ No se encontró el formulario #profileForm en el DOM."); return; } /* ======================================================== 📥 CARGAR DATOS REALES DESDE MONGOOSE (GET) ======================================================== */ try { const res = await fetch(`${API_URL}/api/users/me`, { headers: { Authorization: `Bearer ${token}`, "Content-Type": "application/json" } }); if (!res.ok) throw new Error(`Error del servidor: ${res.status}`); const user = await res.json(); console.log("👤 Datos de perfil recuperados con éxito:", user); // ✏️ Rellenar inputs de texto tradicionales form.firstName.value = user.firstName || ""; form.lastName.value = user.lastName || ""; form.username.value = user.username || ""; form.email.value = user.email || ""; form.age.value = user.age || ""; form.phone.value = user.phone || ""; form.bio.value = user.bio || ""; // 🌎 Rellenar Selectores Dropdown de forma limpia if (user.nationality) form.nationality.value = user.nationality; if (user.department) form.department.value = user.department; if (user.personality) form.personality.value = user.personality; // 📻 Rellenar opción única de tipo Radio Button (Estilo social) if (user.style) { const radioElement = form.querySelector(`input[name="style"][value="${user.style}"]`); if (radioElement) radioElement.checked = true; } // 🎯 Rellenar array de Checkboxes de Intereses if (user.interests && Array.isArray(user.interests)) { user.interests.forEach(interest => { const checkbox = form.querySelector(`input[name="interests"][value="${interest}"]`); if (checkbox) checkbox.checked = true; }); } // 🗣️ Rellenar array de Checkboxes de Idiomas if (user.languages && Array.isArray(user.languages)) { user.languages.forEach(language => { const checkbox = form.querySelector(`input[name="languages"][value="${language}"]`); if (checkbox) checkbox.checked = true; }); } } catch (error) { console.error("❌ Error pintando datos en el formulario:", error); alert("Hubo un problema al cargar tus datos desde el servidor."); } /* ======================================================== ✏️ ENVIAR ACTUALIZACIÓN AL BACKEND (PUT) ======================================================== */ form.addEventListener("submit", async (e) => { e.preventDefault(); // Recolectar dinámicamente los estados checks activos de la pantalla const selectedInterests = Array.from( form.querySelectorAll("input[name='interests']:checked") ).map(el => el.value); const selectedLanguages = Array.from( form.querySelectorAll("input[name='languages']:checked") ).map(el => el.value); // Empaquetar todo el HTMLFormElement en un FormData nativo para Cloudinary/Multer const formData = new FormData(form); // Serializar los arrays para evitar conflictos con el enrutador de Express formData.set("interests", JSON.stringify(selectedInterests)); formData.set("languages", JSON.stringify(selectedLanguages)); try { const res = await fetch(`${API_URL}/api/users/me`, { method: "PUT", headers: { Authorization: `Bearer ${token}` // NOTA CRÍTICA: Nunca definas "Content-Type" cuando mandas un FormData nativo }, body: formData }); const updatedUser = await res.json(); if (!res.ok) { throw new Error(updatedUser.message || "Error al procesar la actualización"); } // 🔄 Sincronizar LocalStorage para reflejar los cambios en el Header/Navbar al instante const updatedSession = { ...authUser, user: { ...authUser.user, firstName: updatedUser.firstName, lastName: updatedUser.lastName, phone: updatedUser.phone, profileImage: updatedUser.profileImage || authUser.user?.profileImage } }; localStorage.setItem("currentUser", JSON.stringify(updatedSession)); alert("✅ ¡Tu perfil ha sido actualizado con éxito!"); // Redirección limpia window.location.href = "index.html"; } catch (error) { console.error("❌ Error guardando cambios de perfil:", error); alert(error.message || "Ocurrió un error inesperado al salvar los datos."); } }); });
